00024  * Uses DeBruijin's multiplication to find the

00040         44, 24, 15,  8, 23,  7,  6,  5
00041     };
00042     /* This magic number multiplies so that the least significant bit
00043      * sets bits [58-64] in the result. This is why we shift over 
00044      * by fifty-eight bits before we look into the array.

00046     static const U64 DeBruijn = 0x07EDD5E59A4E28C2;

00048     return INDEX[((bits & -bits) * DeBruijn) >> 58];
